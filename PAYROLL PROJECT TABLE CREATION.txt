

TABLE NAME: COMPANY


CREATE TABLE company
(COMPANY_ID NUMBER(7) CONSTRAINT C_PK PRIMARY KEY,
COMPANY_NAME VARCHAR2(20) CONSTRAINT C_NN NOT NULL,
LOCATION VARCHAR2(15));

DESC COMPANY;



TABLE NAME: COMPANY_DETAIL


CREATE TABLE company_detail
(COMPANY_ID NUMBER(7) PRIMARY KEY,
COMPANY_NAME VARCHAR2(20) NOT NULL,
LOCATION VARCHAR2(15),
MONTHLY_PROFIT NUMBER);



TABLE NAME: DEPARTMENT_DETAILS

CREATE TABLE department_details
(DEPT_ID NUMBER(3) CONSTRAINT D_PK PRIMARY KEY,
DEPT_NAME VARCHAR2(20) CONSTRAINT D_NN NOT NULL,
EXTRA_ALLOWANCE_PER_DAY NUMBER,
LEAVE_DEDUCTION_PER_DAY NUMBER,
APPRAISAL_AMT NUMBER);


TABLE_NAME: DESIGNATION

CREATE TABLE designation
(DESIGN_ID NUMBER(7) CONSTRAINT DS_PK PRIMARY KEY,
DESIGN_NAME VARCHAR2(20) CONSTRAINT DS_NN NOT NULL,
DEPT_ID REFERENCES DEPARTMENT_DETAILS(DEPT_ID));


TABLE NAME: DESIGN_SAL_DETAIL

CREATE TABLE design_sal_detail 
(DESIGN_ID REFERENCES DESIGNATION(DESIGN_ID),
BASIC_SAL NUMBER CONSTRAINT DD_NN NOT NULL,
DA NUMBER,
TA NUMBER,
HRA NUMBER,
TAX NUMBER,
PPF NUMBER,
DESIGN_NET_SAL NUMBER);



TABLE NAME: BANK_DETAIL

CREATE TABLE bank_detail
(IFSC_CODE VARCHAR2(15) CONSTRAINT B_PK PRIMARY KEY,
BANK_NAME VARCHAR2(20) CONSTRAINT B_NN NOT NULL,
BRANCH_NAME VARCHAR2(15));



TABLE NAME: EMPLOYEE_PERSONAL_DETAIL

CREATE TABLE employee_personal_detail
(EMP_ID NUMBER(6) PRIMARY KEY,
EMP_NAME VARCHAR2(30) NOT NULL,
PHONE_NO NUMBER(10) NOT NULL,
EMAIL_ID VARCHAR2(30),
ACC_NO NUMBER(15) NOT NULL,
IFSC_CODE REFERENCES BANK_DETAIL(IFSC_CODE),
AADHAR_NO NUMBER(12) NOT NULL,
PAN_NO VARCHAR2(10));







TABLE NAME: EMP_WRK_DETAIL

CREATE TABLE emp_wrk_detail
(EMP_ID REFERENCES EMPLOYEE_PERSONAL_DETAIL(EMP_ID),
COMPANY_ID REFERENCES COMPANY(COMPANY_ID),
DESIGN_ID REFERENCES DESIGNATION(DESIGN_ID),
EXTRA_WRK_DAYS NUMBER,
LEAVE_TAKEN_DAYS NUMBER,
APPRAISED VARCHAR2(3));




TABLE NAME: EMPLOYEE_WRK_DETAIL


CREATE TABLE employee_wrk_detail
(EMP_ID REFERENCES EMPLOYEE_PERSONAL_DETAIL(EMP_ID),
COMPANY_ID REFERENCES COMPANY_DETAIL(COMPANY_ID),
DESIGN_ID REFERENCES DESIGNATION(DESIGN_ID),
HIREDATE DATE,
EXTRA_WRK_DAYS NUMBER,
LEAVE_TAKEN_DAYS NUMBER,
APPRAISED VARCHAR2(3));
